// ==========================================
// REGISTROS GLOBAIS E ESTRUTURAS
// ==========================================
const COORDENADOR_URL = "http://127.0.0.1:17050" // Canal 1 atua como Líder

var (
	startTime    time.Time
	txCount      int
	metricsMutex sync.Mutex
)

// Registro Global de Canais Locais
type SkeenRegistry struct {
	mutex  sync.RWMutex
	chains map[string]*chain
}

var globalRegistry = &SkeenRegistry{
	chains: make(map[string]*chain),
}

var serverOnce sync.Once

// Estruturas do Coordenador (Apenas usadas pelo Canal 1)
type CoordTxState struct {
	ExpectedShards []string
	ReceivedTS     map[string]uint64
	FinalTS        uint64
	IsReady        bool
}

var coordState = struct {
	sync.RWMutex
	txs map[string]*CoordTxState
}{txs: make(map[string]*CoordTxState)}

// ==========================================
// INTEGRAÇÃO COM O FABRIC
// ==========================================

type skeenConsenter struct{}

func New() consensus.Consenter {
	serverOnce.Do(func() {
		go startNetworkServer()
	})
	return &skeenConsenter{}
}

func (s *skeenConsenter) HandleChain(support consensus.ConsenterSupport, metadata *common.Metadata) (consensus.Chain, error) {
	c := &chain{
		support:        support,
		channelID:      support.ChannelID(),
		lamportClock:   0,
		pendingQueue:   make(map[string]*PendingTx),
		localTSHistory: make(map[string]uint64),
	}

	globalRegistry.mutex.Lock()
	globalRegistry.chains[support.ChannelID()] = c
	globalRegistry.mutex.Unlock()

	return c, nil
}

func (s *skeenConsenter) IsChannelMember(joinBlock *common.Block) (bool, error) {
	return true, nil
}

type PendingTx struct {
	TxID          string
	Envelope      *common.Envelope
	TSLocal       uint64
	CrossChannels []string
}

type chain struct {
	support   consensus.ConsenterSupport
	channelID string

	mutex          sync.Mutex
	blockMutex     sync.Mutex
	lamportClock   uint64
	pendingQueue   map[string]*PendingTx
	localTSHistory map[string]uint64
}

func (c *chain) Configure(env *common.Envelope, configSeq uint64) error { return nil }
func (c *chain) WaitReady() error                                       { return nil }
func (c *chain) Errored() <-chan struct{}                               { return nil }

func (c *chain) Start() {
	fmt.Printf("[SKEEN COORD] Nó iniciado para o canal: %s\n", c.channelID)
	if c.channelID == "canal1" {
		fmt.Println("👑 [SKEEN COORD] Este nó é o COORDENADOR GLOBAL da rede!")
	}
}

func (c *chain) Halt() {
	fmt.Printf("[SKEEN] Nó parado para o canal: %s\n", c.channelID)
}

// ==========================================
// MOTOR DE CONSENSO (A PORTA DE ENTRADA)
// ==========================================

func (c *chain) Order(env *common.Envelope, configSeq uint64) error {
	metricsMutex.Lock()
	if txCount == 0 {
		startTime = time.Now()
	}
	txCount++
	metricsMutex.Unlock()

	payload, err := protoutil.UnmarshalPayload(env.Payload)
	if err != nil {
		return err
	}

	chdr, err := protoutil.UnmarshalChannelHeader(payload.Header.ChannelHeader)
	if err != nil {
		return err
	}

	txID := chdr.TxId
	if txID == "" {
		txID = "INTERNAL_TX"
	}

	crossChans := []string{c.channelID}
	if strings.HasPrefix(txID, "CROSS_") {
		partes := strings.Split(txID, "_")
		if len(partes) >= 2 {
			crossChans = strings.Split(partes[1], "-")
		}
	}

	// 🚨 FIREWALL DE PROTEÇÃO (EDGE FILTERING)
	souParticipante := false
	for _, ch := range crossChans {
		if ch == c.channelID {
			souParticipante = true
			break
		}
	}

	if !souParticipante {
		return nil
	}

	c.mutex.Lock()
	if _, exists := c.pendingQueue[txID]; exists {
		c.mutex.Unlock()
		return nil
	}
	if _, exists := c.localTSHistory[txID]; exists {
		c.mutex.Unlock()
		return nil
	}

	c.lamportClock++
	meuTSLocal := c.lamportClock

	c.pendingQueue[txID] = &PendingTx{
		TxID:          txID,
		Envelope:      env,
		TSLocal:       meuTSLocal,
		CrossChannels: crossChans,
	}
	c.mutex.Unlock()

	fmt.Printf("[SKEEN Canal %s] Tx [%s] recebida. TS: %d | Papel: PEER\n", c.channelID, txID, meuTSLocal)

	if len(crossChans) > 1 {
		go c.tryFinalizeTx(txID, meuTSLocal, crossChans)
	} else {
		c.FinalizeAndDeliver(txID, meuTSLocal)
	}

	return nil
}

// ==========================================
// FASE 2: COORDENAÇÃO CENTRALIZADA (O(N))
// ==========================================

func (c *chain) tryFinalizeTx(txID string, maxTSLocal uint64, crossChannels []string) {
	finalTS := maxTSLocal

	// FASE 2.1: Enviar TS Local para o Coordenador
	submitURL := fmt.Sprintf("%s/skeen/submit_ts", COORDENADOR_URL)
	reqSubmit, _ := json.Marshal(map[string]interface{}{
		"tx_id":           txID,
		"channel_id":      c.channelID,
		"ts_local":        maxTSLocal,
		"expected_shards": crossChannels,
	})

	fmt.Printf("📡 [COORDENADOR] Shard %s enviando TS %d ao Líder. Tx [%s]\n", c.channelID, maxTSLocal, txID)
	
	// Tenta enviar o TS até conseguir
	for {
		resp, err := http.Post(submitURL, "application/json", bytes.NewBuffer(reqSubmit))
		if err == nil {
			resp.Body.Close()
			break
		}
		time.Sleep(10 * time.Millisecond)
	}

	// FASE 2.2: Fazer polling no Coordenador aguardando o Consenso Final
	pollURL := fmt.Sprintf("%s/skeen/get_final_ts", COORDENADOR_URL)
	reqPoll, _ := json.Marshal(map[string]interface{}{"tx_id": txID})

	for i := 0; i < 50; i++ {
		resp, err := http.Post(pollURL, "application/json", bytes.NewBuffer(reqPoll))
		if err == nil {
			var result struct {
				FinalTS uint64 `json:"final_ts"`
				Ready   bool   `json:"ready"`
			}
			if err := json.NewDecoder(resp.Body).Decode(&result); err == nil {
				if result.Ready {
					finalTS = result.FinalTS
					fmt.Printf("📩 [COORDENADOR] Shard %s recebeu TS FINAL: %d do Líder. Tx [%s]\n", c.channelID, finalTS, txID)
					resp.Body.Close()
					break
				}
			}
			resp.Body.Close()
		}
		time.Sleep(time.Duration(15*(i+1)) * time.Millisecond) // Backoff progressivo
	}

	c.FinalizeAndDeliver(txID, finalTS)
}

func (c *chain) FinalizeAndDeliver(txID string, finalTS uint64) {
	c.mutex.Lock()

	if finalTS > c.lamportClock {
		c.lamportClock = finalTS
		fmt.Printf("⏱️ [SYNC] Shard %s adiantou o relógio interno para %d\n", c.channelID, c.lamportClock)
	}

	tx, exists := c.pendingQueue[txID]
	if !exists {
		c.mutex.Unlock()
		return
	}

	c.localTSHistory[txID] = tx.TSLocal
	envelopeParaBlockCutter := tx.Envelope
	delete(c.pendingQueue, txID)
	c.mutex.Unlock()

	c.blockMutex.Lock()
	defer c.blockMutex.Unlock()

	batches, _ := c.support.BlockCutter().Ordered(envelopeParaBlockCutter)

	for _, batch := range batches {
		blocoSkeen := c.support.CreateNextBlock(batch)
		c.support.WriteBlock(blocoSkeen, nil)

		metricsMutex.Lock()
		decorrido := time.Since(startTime).Seconds()
		tpsReal := float64(txCount) / decorrido
		fmt.Printf("\n[MÉTRICA SKEEN %s] 🧱 Bloco %d Gravado | Total Txs: %d | Tempo: %.2fs | TPS REAL: %.2f\n\n",
			c.channelID, blocoSkeen.Header.Number, txCount, decorrido, tpsReal)
		metricsMutex.Unlock()
	}
}

// ==========================================
// CAMADA DE REDE (API HTTP)
// ==========================================

func startNetworkServer() {
	fabricPortStr := os.Getenv("ORDERER_GENERAL_LISTENPORT")
	if fabricPortStr == "" {
		fabricPortStr = "7050"
	}
	// A API roda na porta do Fabric + 10000 (ex: 17050)
	skeenPort := fabricPortStr + "0" 
	if fabricPortStr == "7050" {
		skeenPort = "17050"
	} else if fabricPortStr == "8050" {
		skeenPort = "18050"
	} else if fabricPortStr == "9050" {
		skeenPort = "19050"
	} else if fabricPortStr == "10050" {
		skeenPort = "20050"
	}

	// ROTA 1: Receber Timestamps Locais (Executado APENAS pelo Coordenador)
	http.HandleFunc("/skeen/submit_ts", func(w http.ResponseWriter, r *http.Request) {
		var req struct {
			TxID           string   `json:"tx_id"`
			ChannelID      string   `json:"channel_id"`
			TSLocal        uint64   `json:"ts_local"`
			ExpectedShards []string `json:"expected_shards"`
		}
		json.NewDecoder(r.Body).Decode(&req)

		coordState.Lock()
		state, exists := coordState.txs[req.TxID]
		if !exists {
			state = &CoordTxState{
				ExpectedShards: req.ExpectedShards,
				ReceivedTS:     make(map[string]uint64),
				FinalTS:        0,
				IsReady:        false,
			}
			coordState.txs[req.TxID] = state
		}

		// Registra o TS recebido deste Shard
		state.ReceivedTS[req.ChannelID] = req.TSLocal

		// Verifica se já recebeu de todos os Shards envolvidos na Tx
		allReceived := true
		for _, expected := range state.ExpectedShards {
			if _, ok := state.ReceivedTS[expected]; !ok {
				allReceived = false
				break
			}
		}

		// Se recebeu de todos, calcula o máximo e finaliza
		if allReceived && !state.IsReady {
			var max uint64 = 0
			for _, ts := range state.ReceivedTS {
				if ts > max {
					max = ts
				}
			}
			state.FinalTS = max
			state.IsReady = true
			fmt.Printf("👑 [LÍDER] Consenso atingido para Tx [%s]. TS Máximo: %d\n", req.TxID, max)
		}
		coordState.Unlock()

		w.WriteHeader(http.StatusOK)
	})

	// ROTA 2: Consultar TS Final (Executado APENAS pelo Coordenador)
	http.HandleFunc("/skeen/get_final_ts", func(w http.ResponseWriter, r *http.Request) {
		var req struct {
			TxID string `json:"tx_id"`
		}
		json.NewDecoder(r.Body).Decode(&req)

		coordState.RLock()
		state, exists := coordState.txs[req.TxID]
		coordState.RUnlock()

		ready := false
		var finalTS uint64 = 0

		if exists && state.IsReady {
			ready = true
			finalTS = state.FinalTS
		}

		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(map[string]interface{}{
			"final_ts": finalTS,
			"ready":    ready,
		})
	})

	fmt.Printf("[SKEEN NETWORK LAYER] API HTTP do Coordenador ativa na porta: %s\n", skeenPort)
	http.ListenAndServe(":"+skeenPort, nil)
}
